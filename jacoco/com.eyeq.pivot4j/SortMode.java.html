package com.eyeq.pivot4j;

public enum SortMode {

	/** sort hierarchically ascending */
	ASC,
	/** sort hierarchically descending */
	DESC,
	/** sort ascending breaking hierarchy */
	BASC,
	/** sort descending breaking hierarchy */
	BDESC,
	/** perform topcount */
	TOPCOUNT,
	/** perform bottomcount */
	BOTTOMCOUNT;
}
