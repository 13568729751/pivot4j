26  public class HtmlTableBuilderIT extends AbstractIntegrationTestCase {
27  
28  	private static final String RESOURCE_PREFIX = "/com/eyeq/pivot4j/ui/html/";
29  
30  	/**
31  	 * @param name
32  	 * @return
33  	 * @throws IOException
34  	 */
35  	protected String readTestResource(String name) throws IOException {
36  		BufferedReader reader = new BufferedReader(new InputStreamReader(
37  				getClass().getResourceAsStream(name)));
38  
39  		StringWriter writer = new StringWriter();
40  
41  		String line = null;
42  		while ((line = reader.readLine()) != null) {
43  			writer.append(line);
44  			writer.append('\n');
45  		}
46  
47  		reader.close();
48  		writer.flush();
49  		writer.close();
50  
51  		return writer.toString().trim();
52  	}
53  
54  	protected void runTestCase(String name) throws IOException {
55  		String mdx = readTestResource(RESOURCE_PREFIX + name + "-mdx.txt");
56  
57  		PivotModel model = getPivotModel();
58  		model.setMdx(mdx);
59  		model.initialize();
60  
61  		HtmlTableBuilder builder = new HtmlTableBuilder();
62  		builder.setHideSpans(false);
63  		builder.setShowDimensionTitle(true);
64  		builder.setShowParentMembers(true);
65  
66  		HtmlTableModel table = builder.build(model);
67  		table.setBorder(1);
68  
69  		StringWriter writer = new StringWriter();
70  		table.writeHtml(new PrintWriter(writer), 0);
71  		writer.flush();
72  		writer.close();
73  
74  		String result = writer.toString().trim();
75  		String expected = readTestResource(RESOURCE_PREFIX + name
76  				+ "-result.html");
77  
78  		assertEquals(expected, result);
79  	}
