52  	@Test
53  	public void testFindVisibleLevels() throws OlapException {
54  		PlaceLevelsOnAxes transform = getTransform();
55  
56  		Cube cube = getPivotModel().getCube();
57  
58  		Hierarchy promotionMedia = cube.getHierarchies().get("Promotion Media");
59  		Hierarchy product = cube.getHierarchies().get("Product");
60  
61  		List<Level> firstLevels = transform.findVisibleLevels(promotionMedia);
62  		assertThat(
63  				"The number of visible levels in [Promotion Media] hierarchy should be 1",
64  				firstLevels.size(), is(equalTo(1)));
65  		assertThat(
66  				"The first level of the hierarchy should be [Promotion Media].[(All)]",
67  				promotionMedia.getLevels().get(0).getUniqueName(),
68  				is(equalTo(firstLevels.get(0).getUniqueName())));
69  
70  		List<Level> allLevels = transform.findVisibleLevels(Axis.ROWS);
71  
72  		assertThat("Visible levels should not be empty", allLevels.isEmpty(),
73  				is(false));
74  		assertThat("The number of visible levels should be 5",
75  				allLevels.size(), is(equalTo(5)));
76  		assertThat(
77  				"The first level of the axis should be [Promotion Media].[(All)]",
78  				promotionMedia.getLevels().get(0).getUniqueName(),
79  				is(equalTo(allLevels.get(0).getUniqueName())));
80  
81  		assertThat("The second level of the axis should be [Product].[(All)]",
82  				product.getLevels().get(0).getUniqueName(),
83  				is(equalTo(allLevels.get(1).getUniqueName())));
84  		assertThat(
85  				"The third level of the axis should be [Product].[Product Family]",
86  				product.getLevels().get(1).getUniqueName(),
87  				is(equalTo(allLevels.get(2).getUniqueName())));
88  		assertThat(
89  				"The fourth level of the axis should be [Product].[Product Department]",
90  				product.getLevels().get(2).getUniqueName(),
91  				is(equalTo(allLevels.get(3).getUniqueName())));
92  		assertThat(
93  				"The fourth level of the axis should be [Product].[Product Category]",
94  				product.getLevels().get(3).getUniqueName(),
95  				is(equalTo(allLevels.get(4).getUniqueName())));
96  	}
