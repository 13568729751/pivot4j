106 	public void testParseKeyIdentifier() throws Exception {
107 		String mdx = "SELECT [AAA].&[BBB] ON COLUMNS, [CCC].&[DDD].[EEE] ON ROWS FROM DummyCube";
108 
109 		MdxStatement query = parseQuery(mdx);
110 
111 		assertThat("Failed to parse : " + mdx, query, is(notNullValue()));
112 		assertThat("Axes should not be null.", query.getAxes(),
113 				is(notNullValue()));
114 
115 		assertThat("Number of axes should be 2.", query.getAxes().size(),
116 				is(equalTo(2)));
117 
118 		Exp columnExp = query.getAxes().get(0).getExp();
119 
120 		assertThat("Exp object for columns axis is null.", columnExp,
121 				is(notNullValue()));
122 
123 		assertThat("Exp object should be instance of CompoundId.", columnExp,
124 				is(instanceOf(CompoundId.class)));
125 
126 		CompoundId columnId = (CompoundId) columnExp;
127 		assertThat("Wrong number of name parts on column axis member.",
128 				columnId.getNames().size(), is(equalTo(2)));
129 
130 		assertThat("First name part has wrong identifier.", columnId.getNames()
131 				.get(0).getName(), is(equalTo("[AAA]")));
132 		assertThat("Second name part has wrong identifier.", columnId
133 				.getNames().get(1).getName(), is(equalTo("[BBB]")));
134 		assertThat("First name part cannot be a key identifier.", columnId
135 				.getNames().get(0).isKey(), is(false));
136 		assertThat("Second name part is a key identifier.", columnId.getNames()
137 				.get(1).isKey(), is(true));
138 
139 		Exp rowExp = query.getAxes().get(1).getExp();
140 
141 		assertThat("Exp object for rows axis is null.", rowExp,
142 				is(notNullValue()));
143 
144 		assertThat("Exp object should be instance of CompoundId.", rowExp,
145 				is(instanceOf(CompoundId.class)));
146 
147 		CompoundId rowId = (CompoundId) rowExp;
148 		assertThat("Wrong number of name parts on row axis member.", rowId
149 				.getNames().size(), is(equalTo(3)));
150 
151 		assertThat("First name part has wrong identifier.", rowId.getNames()
152 				.get(0).getName(), is(equalTo("[CCC]")));
153 		assertThat("Second name part has wrong identifier.", rowId.getNames()
154 				.get(1).getName(), is(equalTo("[DDD]")));
155 		assertThat("Third name part has wrong identifier.", rowId.getNames()
156 				.get(2).getName(), is(equalTo("[EEE]")));
157 		assertThat("First name part cannot be a key identifier.", rowId
158 				.getNames().get(0).isKey(), is(false));
159 		assertThat("Second name part is a key identifier.", rowId.getNames()
160 				.get(1).isKey(), is(true));
161 		assertThat("Third name part is not a key identifier.", rowId.getNames()
162 				.get(2).isKey(), is(false));
163 	}
