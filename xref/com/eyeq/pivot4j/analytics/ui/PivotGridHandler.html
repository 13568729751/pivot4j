205 			defaultItem.setItemLabel(defaultLabel);

207 
208 			cubeItems.add(defaultItem);
209 
210 			if (model != null) {
211 				OlapDataSource dataSource = ((PivotModelImpl) model)
212 						.getDataSource();
213 
214 				Schema schema = dataSource.getConnection().getOlapSchema();
215 
216 				List<Cube> cubes = schema.getCubes();
217 
218 				for (Cube cube : cubes) {
219 					UISelectItem item = new UISelectItem();
220 					item.setItemLabel(cube.getCaption());
221 					item.setItemValue(cube.getName());
222 
223 					cubeItems.add(item);
224 				}
225 			}
226 		}
227 
228 		return cubeItems;

240 			model.setMdx(mdx);

242 			if (!model.isInitialized()) {

248 	public boolean isValid() {
249 		if (model == null || !model.isInitialized()) {
250 			return false;
251 		}
252 
253 		List<CellSetAxis> axes = model.getCellSet().getAxes();
254 		if (axes.size() < 2) {
255 			return false;
256 		}
257 
258 		return axes.get(0).getPositionCount() > 0
259 				&& axes.get(1).getPositionCount() > 0;
260 	}

304 					.get("hierarchy")));

309 					.get("cell")));

313 				.get("command"));
