210 	public void startCell(TableRenderContext context) {
211 		this.column = new Column();
212 
213 		String id = "col-" + column.hashCode();
214 
215 		column.setId(id);
216 		column.setColspan(context.getColumnSpan());
217 		column.setRowspan(context.getRowSpan());
218 
219 		RenderPropertyUtils propertyUtils = getRenderPropertyUtils();
220 
221 		String propertyCategory = context.getRenderPropertyCategory();
222 
223 		StringWriter writer = new StringWriter();
224 		CssWriter cssWriter = new CssWriter(writer);
225 
226 		String type = context.getCellType();
227 
228 		if (type.equals(LABEL) && !context.getRenderer().getShowParentMembers()
229 				&& context.getMember() != null) {
230 			int padding = context.getMember().getDepth() * 10;
231 			cssWriter.writeStyle("padding-left", padding + "px");
232 		}
233 
234 		String fgColor = propertyUtils.getString("fgColor", propertyCategory,
235 				null);
236 
237 		if (fgColor != null) {
238 			cssWriter.writeStyle("color", fgColor);
239 		}
240 
241 		String bgColor = propertyUtils.getString("bgColor", propertyCategory,
242 				null);
243 
244 		if (bgColor != null) {
245 			cssWriter.writeStyle("background-color", bgColor);
246 			cssWriter.writeStyle("background-image", "none");
247 		}
248 
249 		String fontFamily = propertyUtils.getString("fontFamily",
250 				propertyCategory, null);
251 
252 		if (fontFamily != null) {
253 			cssWriter.writeStyle("font-family", fontFamily);
254 		}
255 
256 		String fontSize = propertyUtils.getString("fontSize", propertyCategory,
257 				null);
258 
259 		if (fontSize != null) {
260 			cssWriter.writeStyle("font-size", fontSize);
261 		}
262 
263 		String fontStyle = propertyUtils.getString("fontStyle",
264 				propertyCategory, null);
265 
266 		if (fontStyle != null) {
267 			if (fontStyle.contains("bold")) {
268 				cssWriter.writeStyle("font-weight", "bold");
269 			}
270 
271 			if (fontStyle.contains("italic")) {
272 				cssWriter.writeStyle("font-style", "oblique");
273 			}
274 		}
275 
276 		writer.flush();
277 
278 		IOUtils.closeQuietly(writer);
279 
280 		String style = writer.toString();
281 
282 		if (StringUtils.isNotEmpty(style)) {
283 			column.setStyle(style);
284 		}
285 
286 		String styleClass = getStyleClass(context);
287 		String styleClassProperty = propertyUtils.getString("styleClass",
288 				propertyCategory, null);
289 
290 		if (styleClassProperty != null) {
291 			if (styleClass == null) {
292 				styleClass = styleClassProperty;
293 			} else {
294 				styleClass += " " + styleClassProperty;
295 			}
296 		}
297 
298 		column.setStyleClass(styleClass);
299 	}

301 	/**
302 	 * @param context
303 	 * @return
304 	 */
305 	protected String getStyleClass(TableRenderContext context) {
306 		String styleClass = null;
307 
308 		String type = context.getCellType();
309 
310 		if (type.equals(LABEL) || type.equals(TITLE) || type.equals(FILL)) {
311 			if (context.getAxis() == Axis.COLUMNS) {
312 				styleClass = "col-hdr-cell";
313 			} else {
314 				if (type.equals(LABEL)) {
315 					styleClass = "row-hdr-cell ui-widget-header";
316 				} else {
317 					styleClass = "ui-widget-header";
318 				}
319 			}
320 		} else if (type.equals(AGG_VALUE)) {
321 			if (context.getAxis() == Axis.ROWS) {
322 				styleClass = "ui-widget-header ";
323 			} else {
324 				styleClass = "";
325 			}
326 
327 			styleClass += "agg-title";
328 		} else if (type.equals(VALUE)) {
329 			if (context.getAggregator() == null) {
330 				// PrimeFaces' Row class doesn't have the styleClass property.
331 				if (context.getRowIndex() % 2 == 0) {
332 					styleClass = "value-cell cell-even";
333 				} else {
334 					styleClass = "value-cell cell-odd";
335 				}
336 			} else {
337 				styleClass = "ui-widget-header agg-cell";
338 
339 				if (context.getAxis() == Axis.COLUMNS) {
340 					styleClass += " col-agg-cell";
341 				} else if (context.getAxis() == Axis.ROWS) {
342 					styleClass += " row-agg-cell";
343 				}
344 			}
345 		}
346 
347 		return styleClass;
348 	}

439 		} finally {
440 			context.getExpressionContext().remove("label");
441 		}
442 
443 		HtmlOutputText text = new HtmlOutputText();
444 		String id = "txt-" + text.hashCode();
445 
446 		text.setId(id);
447 		text.setValue(labelText);
448 
449 		String link = propertyUtils.getString("link",
450 				context.getRenderPropertyCategory(), null);
451 
452 		if (link == null) {
453 			column.getChildren().add(text);
454 		} else {
455 			HtmlOutputLink anchor = new HtmlOutputLink();
456 			anchor.setValue(link);
457 			anchor.getChildren().add(text);
458 
459 			column.getChildren().add(anchor);

513 		for (String category : context.getRenderProperties().keySet()) {
514 			Map<String, EvaluationFailedException> errors = getRenderPropertyUtils()
515 					.getLastErrors(category);
516 
517 			for (String property : errors.keySet()) {
518 				String title = mf.format(new String[] { resources
519 						.getString("properties." + property) });
520 
521 				EvaluationFailedException e = errors.get(property);
522 
523 				facesContext.addMessage(null, new FacesMessage(
524 						FacesMessage.SEVERITY_ERROR, title, e.getMessage()));
525 
526 				if (logger.isWarnEnabled()) {
527 					logger.warn(title, e);
528 				}
529 			}
530 		}
531 
532 		this.commandIndex = 0;
