32  	@Override
33  	public void decode(FacesContext context, UIComponent component) {
34  		decodeBehaviors(context, component);
35  
36  		super.decode(context, component);
37  	}

39  	/**
40  	 * @see org.primefaces.component.colorpicker.ColorPickerRenderer#encodeScript(javax.faces.context.FacesContext,
41  	 *      org.primefaces.component.colorpicker.ColorPicker)
42  	 */
43  	@Override
44  	protected void encodeScript(FacesContext context, ColorPicker colorPicker)
45  			throws IOException {
46  		ResponseWriter writer = context.getResponseWriter();
47  		String clientId = colorPicker.getClientId(context);
48  		String value = (String) colorPicker.getValue();
49  		WidgetBuilder wb = getWidgetBuilder(context);
50  
51  		wb.widget("AjaxColorPicker", colorPicker.resolveWidgetVar(), clientId,
52  				"colorpicker", true).attr("mode", colorPicker.getMode())
53  				.attr("color", value, null);
54  
55  		encodeClientBehaviors(context, (ClientBehaviorHolder) colorPicker, wb);
56  
57  		startScript(writer, clientId);
58  		writer.write(wb.build());
59  
60  		endScript(writer);
61  	}
