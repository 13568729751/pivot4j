9   package com.eyeq.pivot4j.analytics.component;
10  
11  import java.util.Arrays;
12  import java.util.Collection;
13  import java.util.Map;
14  
15  import javax.faces.application.ResourceDependencies;
16  import javax.faces.application.ResourceDependency;
17  import javax.faces.component.FacesComponent;
18  import javax.faces.component.behavior.ClientBehaviorHolder;
19  import javax.faces.context.FacesContext;
20  import javax.faces.event.AjaxBehaviorEvent;
21  import javax.faces.event.FacesEvent;
22  
23  import org.primefaces.component.colorpicker.ColorPicker;
24  import org.primefaces.util.Constants;
25  
26  /**
27   * Workaround for lack of client behavior support in PrimeFaces' ColorPicker
28   * component.
29   */
30  @FacesComponent("com.eyeq.pivot4j.component.ColorPicker")
31  @ResourceDependencies({ @ResourceDependency(library = "pivot4j", name = "js/colorpicker.js") })
32  public class AjaxColorPicker extends ColorPicker implements
33  		ClientBehaviorHolder {
34  
35  	public static final String COMPONENT_FAMILY = "com.eyeq.pivot4j.component";
36  
37  	public static final String COMPONENT_TYPE = "com.eyeq.pivot4j.component.ColorPicker";
38  
39  	public static final String RENDERER_TYPE = "com.eyeq.pivot4j.component.ColorPickerRenderer";
40  
41  	public AjaxColorPicker() {
42  		setRendererType(RENDERER_TYPE);
43  	}
44  
45  	/**
46  	 * @see org.primefaces.component.colorpicker.ColorPicker#getFamily()
47  	 */
48  	@Override
49  	public String getFamily() {
50  		return COMPONENT_FAMILY;
51  	}
52  
53  	/**
54  	 * @see javax.faces.component.UIComponentBase#getEventNames()
55  	 */
56  	@Override
57  	public Collection<String> getEventNames() {
58  		return Arrays.asList("change");
59  	}
60  
61  	/**
62  	 * @see javax.faces.component.UIComponentBase#queueEvent(javax.faces.event.FacesEvent)
63  	 */
64  	@Override
65  	public void queueEvent(FacesEvent event) {
66  		FacesContext context = getFacesContext();
67  
68  		if (event instanceof AjaxBehaviorEvent) {
69  			AjaxBehaviorEvent behaviorEvent = (AjaxBehaviorEvent) event;
70  			Map<String, String> params = context.getExternalContext()
71  					.getRequestParameterMap();
72  			String eventName = params
73  					.get(Constants.PARTIAL_BEHAVIOR_EVENT_PARAM);
74  			String clientId = this.getClientId(context);
75  
76  			if (eventName.equals("change")) {
77  				String value = params.get(clientId + "_input");
78  
79  				if (value != null) {
80  					ChangeEvent changeEvent = new ChangeEvent(this,
81  							behaviorEvent.getBehavior(), value);
82  					changeEvent.setPhaseId(behaviorEvent.getPhaseId());
83  
84  					super.queueEvent(changeEvent);
85  				}
86  			}
87  		} else {
88  			super.queueEvent(event);
89  		}
90  	}
91  }
