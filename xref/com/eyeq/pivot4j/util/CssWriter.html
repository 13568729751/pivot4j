9   package com.eyeq.pivot4j.util;
10  
11  import java.io.PrintWriter;
12  import java.io.Writer;
13  
14  import org.apache.commons.lang.NullArgumentException;
15  import org.apache.commons.lang.StringUtils;
16  
17  public class CssWriter {
18  
19  	private PrintWriter writer;
20  
21  	/**
22  	 * @param writer
23  	 */
24  	public CssWriter(Writer writer) {
25  		if (writer == null) {
26  			throw new NullArgumentException("writer");
27  		}
28  
29  		this.writer = new PrintWriter(writer);
30  	}
31  
32  	/**
33  	 * @return the writer
34  	 */
35  	public PrintWriter getWriter() {
36  		return writer;
37  	}
38  
39  	/**
40  	 * @param name
41  	 * @param value
42  	 */
43  	public void writeStyle(String name, String value) {
44  		if (StringUtils.isNotBlank(value)) {
45  			writer.print(name);
46  			writer.print(": ");
47  			writer.print(value);
48  			writer.print(";");
49  		}
50  	}
51  }
