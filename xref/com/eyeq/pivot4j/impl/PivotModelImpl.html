317 	/**
318 	 * @see com.eyeq.pivot4j.PivotModel#getCellSet()
319 	 */
320 	public synchronized CellSet getCellSet() throws NotInitializedException {
321 		checkInitialization();
322 
323 		if (cellSet != null) {
324 			return cellSet;
325 		}
326 
327 		long t1 = System.currentTimeMillis();
328 
329 		if (queryAdapter == null) {
330 			throw new IllegalStateException("Initial MDX is not specified.");
331 		}
332 
333 		String mdx = normalizeMdx(getCurrentMdx());
334 
335 		if (logger.isDebugEnabled()) {
336 			logger.debug(mdx);
337 		}
338 
339 		try {
340 			this.cellSet = executeMdx(connection, mdx);
341 
342 			Cube cube = cellSet.getMetaData().getCube();
343 
344 			this.cubeName = cube.getName();
345 			this.schemaName = cube.getSchema().getName();
346 		} catch (OlapException e) {
347 			throw new PivotException(e);
348 		}
349 
350 		queryAdapter.afterExecute(cellSet);
351 
352 		if (logger.isInfoEnabled()) {
353 			long t2 = System.currentTimeMillis();
354 			logger.info("Query execution time " + (t2 - t1) + " ms");
355 		}
356 
357 		return cellSet;
358 	}

406 	public void setMdx(String mdxQuery) {
407 		if (mdxQuery == null) {
408 			throw new IllegalArgumentException("MDX query cannot be null.");
409 		}
410 
411 		this.mdxQuery = mdxQuery;
412 
413 		String mdx = normalizeMdx(mdxQuery);
414 		if (!mdx.equals(normalizeMdx(getCurrentMdx()))) {
415 			onMdxChanged(mdx);
416 		}

419 	protected void onMdxChanged(String mdx) {
420 		if (logger.isInfoEnabled()) {
421 			logger.info("setMdx: " + mdx);
422 		}
423 
424 		this.cellSet = null;
425 		this.topBottomCount = 10;
426 		this.sortMode = SortMode.ASC;
427 		this.sorting = false;
428 		this.sortPosMembers = null;
429 
430 		if (queryAdapter != null) {
431 			queryAdapter.removeChangeListener(queryChangeListener);
432 			queryAdapter = null;
433 		}
434 
435 		if (isInitialized()) {
436 			this.queryAdapter = createQueryAdapter();
437 			queryAdapter.updateQuery();
438 
439 			queryAdapter.addChangeListener(queryChangeListener);
440 		}
441 	}

519 	/**
520 	 * @param position
521 	 *            to be checked
522 	 * @return true, if position is the current sorting position
523 	 * @see com.eyeq.pivot4j.PivotModel#isSorting(org.olap4j.Position)
524 	 */
525 	public boolean isSorting(Position position) {
526 		if (!isSortOnQuery()) {
527 			return false;
528 		} else {
529 			if (sortPosMembers.size() != position.getMembers().size()) {
530 				return false;
531 			}
532 
533 			for (int i = 0; i < sortPosMembers.size(); i++) {
534 				Member member1 = sortPosMembers.get(i);
535 				Member member2 = position.getMembers().get(i);
536 				// any null does not compare
537 				if (member1 == null) {
538 					return false;
539 				} else if (!member1.equals(member2)) {
540 					return false;
541 				}
542 			}
543 			return true;
544 		}
545 	}

595 	 * 
596 	 * @param position
597 	 *            the position to check for sortability
598 	 * @return true, if the position is sortable
599 	 */
600 	public boolean isSortable(Position position) {
601 		try {
602 			List<Member> members = position.getMembers();
603 			for (Member member : members) {
604 				if (member.getLevel().getHierarchy().getDimension()
605 						.getDimensionType() == Type.MEASURE) {
606 					return true;
607 				}
608 			}
609 		} catch (OlapException e) {
610 			throw new PivotException(e);
611 		}
612 
613 		return false;
614 	}

660 	/**
661 	 * @param membersToSort
662 	 *            Axis containing the members to be sorted
663 	 * @param position
664 	 *            Position on "other axis" defining the members by which the
665 	 *            membersToSort are sorted
666 	 */
667 	public void sort(CellSetAxis membersToSort, Position position) {
668 		List<Position> positions = membersToSort.getPositions();
669 
670 		// if the axis to sort does not contain any positions - sorting is not
671 		// posssible
672 		if (positions.isEmpty()) {
673 			if (logger.isWarnEnabled()) {
674 				logger.warn("Reject sort, the axis to be sorted is empty.");
675 			}
676 
677 			this.sorting = false;
678 			return;
679 		}
680 
681 		this.sortPosMembers = position.getMembers();
682 
683 		// find the axis to sort
684 		Dimension dim = positions.get(0).getMembers().get(0).getDimension();
685 
686 		Quax quaxToSort = getQueryAdapter().findQuax(dim);
687 
688 		if (quaxToSort == null) {
689 			if (logger.isWarnEnabled()) {
690 				logger.warn("Reject sort, the Quax is null");
691 			}
692 			this.sorting = false;
693 			return;
694 		}
695 
696 		getQueryAdapter().setQuaxToSort(quaxToSort);
697 
698 		if (logger.isInfoEnabled()) {
699 			StringBuilder builder = new StringBuilder();
700 			builder.append("Change Sort Position ");
701 
702 			boolean first = true;
703 
704 			List<Member> members = position.getMembers();
705 			for (Member member : members) {
706 				if (first) {
707 					first = false;
708 				} else {
709 					builder.append(" ");
710 				}
711 				builder.append(member.getUniqueName());
712 			}
713 			builder.append(" iAxisToSort=");
714 			builder.append(Integer.toString(quaxToSort.getOrdinal()));
715 
716 			logger.info(builder.toString());
717 		}
718 
719 		fireModelChanged();
720 	}

722 	/**
723 	 * @see com.eyeq.pivot4j.StateHolder#bookmarkState()
724 	 */
725 	public synchronized Serializable bookmarkState() {
726 		Serializable[] state = new Serializable[5];
727 
728 		state[0] = getCurrentMdx();
729 		state[1] = cubeName;
730 		state[2] = schemaName;
731 
732 		if (sortPosMembers == null) {
733 			state[3] = null;
734 		} else {
735 			Serializable[] sortState = new Serializable[4];
736 
737 			String[] uniqueNames = new String[sortPosMembers.size()];
738 			for (int i = 0; i < uniqueNames.length; i++) {
739 				uniqueNames[i] = sortPosMembers.get(i).getUniqueName();
740 			}
741 
742 			sortState[0] = uniqueNames;
743 			sortState[1] = getTopBottomCount();
744 			sortState[2] = getSortMode();
745 			sortState[3] = isSorting();
746 
747 			state[3] = sortState;
748 		}
749 
750 		state[4] = getQueryAdapter().bookmarkState();
751 
752 		return state;
753 	}

755 	/**
756 	 * @see com.eyeq.pivot4j.StateHolder#restoreState(java.io.Serializable)
757 	 */
758 	public synchronized void restoreState(Serializable state) {
759 		Serializable[] states = (Serializable[]) state;
760 
761 		setMdx((String) states[0]);
762 
763 		this.cubeName = (String) states[1];
764 		this.schemaName = (String) states[2];
765 
766 		if (!isInitialized()) {
767 			initialize();
768 		}
769 
770 		// sorting
771 		if (states[3] == null || cubeName == null || schemaName == null) {
772 			this.sortPosMembers = null;
773 		} else {
774 			try {
775 				Schema schema = getCatalog().getSchemas().get(schemaName);
776 				Cube cube = schema.getCubes().get(cubeName);
777 
778 				Serializable[] sortStates = (Serializable[]) states[3];
779 
780 				String[] sortPosUniqueNames = (String[]) sortStates[0];
781 				if (sortPosUniqueNames == null) {
782 					this.sortPosMembers = null;
783 				} else {
784 					this.sortPosMembers = new ArrayList<Member>(
