9   package com.eyeq.pivot4j.mdx.impl;

11  import java.io.StringReader;

13  import java_cup.runtime.Symbol;
14  
15  import com.eyeq.pivot4j.PivotException;
16  import com.eyeq.pivot4j.mdx.MdxParser;
17  import com.eyeq.pivot4j.mdx.MdxStatement;
18  import com.eyeq.pivot4j.mdx.ParseException;
19  
20  public class MdxParserImpl implements MdxParser {
21  
22  	/**
23  	 * @see com.eyeq.pivot4j.mdx.MdxParser#parse(java.lang.String)
24  	 */
25  	@Override
26  	public MdxStatement parse(String mdx) {
27  		if (mdx == null) {
28  			throw new IllegalArgumentException(
29  					"MDX query argument cannot be null.");
30  		}
31  
32  		MdxStatement query;
33  
34  		try {
35  			CupParser parser = new CupParser(new StringReader(mdx));
36  			Symbol parseTree = parser.parse();
37  
38  			query = (MdxStatement) parseTree.value;
39  		} catch (PivotException e) {
40  			throw e;
41  		} catch (Exception e) {
42  			String msg = "Failed to parse MDX query : " + mdx;
43  			throw new ParseException(msg, e);
44  		}
45  
46  		return query;
47  	}
48  }
