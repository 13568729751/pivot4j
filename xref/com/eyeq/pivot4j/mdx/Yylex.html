436   /**
437    * Refills the input buffer.
438    *
439    * @return      <code>false</code>, iff there was new input.
440    * 
441    * @exception   java.io.IOException  if any I/O-Error occurs
442    */
443   private boolean zzRefill() throws java.io.IOException {
444 
445     /* first: make room (if you can) */
446     if (zzStartRead > 0) {
447       System.arraycopy(zzBuffer, zzStartRead,
448                        zzBuffer, 0,
449                        zzEndRead-zzStartRead);
450 
451       /* translate stored positions */
452       zzEndRead-= zzStartRead;
453       zzCurrentPos-= zzStartRead;
454       zzMarkedPos-= zzStartRead;
455       zzStartRead = 0;
456     }
457 
458     /* is the buffer big enough? */
459     if (zzCurrentPos >= zzBuffer.length) {
460       /* if not: blow it up */
461       char newBuffer[] = new char[zzCurrentPos*2];
462       System.arraycopy(zzBuffer, 0, newBuffer, 0, zzBuffer.length);
463       zzBuffer = newBuffer;
464     }
465 
466     /* finally: fill the buffer with new input */
467     int numRead = zzReader.read(zzBuffer, zzEndRead,
468                                             zzBuffer.length-zzEndRead);
469 
470     if (numRead > 0) {
471       zzEndRead+= numRead;
472       return false;
473     }
474     // unlikely but not impossible: read 0 characters, but not at end of stream    
475     if (numRead == 0) {
476       int c = zzReader.read();
477       if (c == -1) {
478         return true;
479       } else {
480         zzBuffer[zzEndRead++] = (char) c;
481         return false;
482       }     
483     }
484 
485 	// numRead < 0
486     return true;
487   }
