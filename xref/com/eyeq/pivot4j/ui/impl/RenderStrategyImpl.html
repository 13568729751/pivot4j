51  	/**
52  	 * @param model
53  	 * @param renderer
54  	 * @param callback
55  	 * @see com.eyeq.pivot4j.ui.RenderStrategy#render(com.eyeq.pivot4j.PivotModel,
56  	 *      com.eyeq.pivot4j.ui.PivotRenderer,
57  	 *      com.eyeq.pivot4j.ui.PivotLayoutCallback)
58  	 */
59  	public void render(PivotModel model, PivotRenderer renderer,
60  			PivotLayoutCallback callback) {
61  		if (model == null) {
62  			throw new NullArgumentException("model");
63  		}
64  
65  		if (renderer == null) {
66  			throw new NullArgumentException("renderer");
67  		}
68  
69  		List<CellSetAxis> axes = model.getCellSet().getAxes();
70  		if (axes.isEmpty()) {
71  			return;
72  		}
73  
74  		TableHeaderNode columnRoot = createAxisTree(model, renderer,
75  				Axis.COLUMNS);
76  		if (columnRoot == null) {
77  			return;
78  		}
79  
80  		TableHeaderNode rowRoot = createAxisTree(model, renderer, Axis.ROWS);
81  		if (rowRoot == null) {
82  			return;
83  		}
84  
85  		configureAxisTree(model, renderer, Axis.COLUMNS, columnRoot);
86  		configureAxisTree(model, renderer, Axis.ROWS, rowRoot);
87  
88  		invalidateAxisTree(model, Axis.COLUMNS, columnRoot);
89  		invalidateAxisTree(model, Axis.ROWS, rowRoot);
90  
91  		RenderContext context = createRenderContext(model, renderer,
92  				columnRoot, rowRoot);
93  
94  		callback.startTable(context);
95  
96  		renderHeader(context, columnRoot, rowRoot, callback);
97  		renderBody(context, columnRoot, rowRoot, callback);
98  
99  		callback.endTable(context);
100 	}

109 	protected RenderContext createRenderContext(PivotModel model,
110 			PivotRenderer renderer, TableHeaderNode columnRoot,
111 			TableHeaderNode rowRoot) {
112 		int columnHeaderCount = columnRoot.getMaxRowIndex();
113 		int rowHeaderCount = rowRoot.getMaxRowIndex();
114 
115 		int columnCount = columnRoot.getWidth();
116 		int rowCount = rowRoot.getWidth();
117 
118 		Map<String, Member> cachedParents = new HashMap<String, Member>();
119 
120 		cachedParents.putAll(columnRoot.getReference().getParentMembersCache());
121 		cachedParents.putAll(rowRoot.getReference().getParentMembersCache());
122 
123 		return new RenderContext(model, renderer, columnCount, rowCount,
124 				columnHeaderCount, rowHeaderCount, cachedParents);
125 	}

127 	/**
128 	 * @param context
129 	 * @param columnRoot
130 	 * @param rowRoot
131 	 * @param callback
132 	 */
133 	protected void renderHeader(final RenderContext context,
134 			final TableHeaderNode columnRoot, final TableHeaderNode rowRoot,
135 			final PivotLayoutCallback callback) {
136 
137 		callback.startHeader(context);
138 
139 		int count = context.getColumnHeaderCount();
140 
141 		for (int rowIndex = 0; rowIndex < count; rowIndex++) {
142 			context.setAxis(Axis.COLUMNS);
143 			context.setColIndex(0);
144 			context.setRowIndex(rowIndex);
145 
146 			callback.startRow(context);
147 
148 			renderHeaderCorner(context, columnRoot, rowRoot, callback);
149 
150 			// invoking renderHeaderCorner method resets the axis property.
151 			context.setAxis(Axis.COLUMNS);
152 
153 			columnRoot.walkChildrenAtRowIndex(
154 					new TreeNodeCallback<TableAxisContext>() {
155 
156 						@Override
157 						public int handleTreeNode(
158 								TreeNode<TableAxisContext> node) {
159 							TableHeaderNode headerNode = (TableHeaderNode) node;
160 
161 							context.setColIndex(headerNode.getColIndex()
162 									+ context.getRowHeaderCount());
163 							context.setColSpan(headerNode.getColSpan());
164 							context.setRowSpan(headerNode.getRowSpan());
165 
166 							context.setMember(headerNode.getMember());
167 							context.setProperty(headerNode.getProperty());
168 							context.setHierarchy(headerNode.getHierarchy());
169 							context.setColumnPosition(headerNode.getPosition());
170 							context.setAggregator(headerNode.getAggregator());
171 							context.setCell(null);
172 
173 							if (headerNode.isAggregation()) {
174 								context.setCellType(CellType.Aggregation);
175 							} else if (context.getMember() == null) {
176 								if (context.getHierarchy() == null) {
177 									context.setCellType(CellType.None);
178 								} else {
179 									context.setCellType(CellType.Title);
180 								}
181 							} else {
182 								context.setCellType(CellType.Header);
183 							}
184 
185 							callback.startCell(context);
186 							callback.cellContent(context);
187 							callback.endCell(context);
188 
189 							return TreeNodeCallback.CONTINUE;
190 						}
191 					}, rowIndex + 1);
192 
193 			callback.endRow(context);
194 		}
195 
196 		callback.endHeader(context);
197 	}

199 	/**
200 	 * @param context
201 	 * @param rowRoot
202 	 * @param columnRoot
203 	 * @param callback
204 	 */
205 	protected void renderBody(final RenderContext context,
206 			final TableHeaderNode columnRoot, final TableHeaderNode rowRoot,
207 			final PivotLayoutCallback callback) {
208 		callback.startBody(context);
209 
210 		int count = rowRoot.getColSpan();
211 
212 		for (int rowIndex = 0; rowIndex < count; rowIndex++) {
213 			context.setAxis(Axis.ROWS);
214 			context.setColIndex(0);
215 			context.setRowIndex(rowIndex + context.getColumnHeaderCount());
216 
217 			callback.startRow(context);
218 
219 			rowRoot.walkChildrenAtColIndex(
220 					new TreeNodeCallback<TableAxisContext>() {
221 
222 						@Override
223 						public int handleTreeNode(
224 								TreeNode<TableAxisContext> node) {
225 							TableHeaderNode headerNode = (TableHeaderNode) node;
226 
227 							if (headerNode.getRowIndex() == 0) {
228 								return TreeNodeCallback.CONTINUE;
229 							}
230 
231 							context.setColIndex(headerNode.getRowIndex() - 1);
232 							context.setColSpan(headerNode.getRowSpan());
233 							context.setRowSpan(headerNode.getColSpan());
234 
235 							context.setMember(headerNode.getMember());
236 							context.setProperty(headerNode.getProperty());
237 							context.setHierarchy(headerNode.getHierarchy());
238 							context.setRowPosition(headerNode.getPosition());
239 							context.setCell(null);
240 							context.setAggregator(headerNode.getAggregator());
241 
242 							if (headerNode.isAggregation()) {
243 								context.setCellType(CellType.Aggregation);
244 							} else if (context.getMember() == null) {
245 								if (context.getHierarchy() == null) {
246 									context.setCellType(CellType.None);
247 								} else {
248 									context.setCellType(CellType.Title);
249 								}
250 							} else {
251 								context.setCellType(CellType.Header);
252 							}
253 
254 							callback.startCell(context);
255 							callback.cellContent(context);
256 							callback.endCell(context);
257 
258 							if (headerNode.getChildCount() == 0) {
259 								renderDataRow(context, columnRoot, rowRoot,
260 										(TableHeaderNode) node, callback);
261 							}
262 
263 							return TreeNodeCallback.CONTINUE;
264 						}
265 					}, rowIndex);
266 
267 			callback.endRow(context);
268 		}
269 
270 		callback.endBody(context);
271 	}

273 	/**
274 	 * @param context
275 	 * @param columnRoot
276 	 * @param rowRoot
277 	 * @param rowNode
278 	 * @param callback
279 	 */
280 	protected void renderDataRow(RenderContext context,
281 			TableHeaderNode columnRoot, TableHeaderNode rowRoot,
282 			TableHeaderNode rowNode, PivotLayoutCallback callback) {
283 		context.setCellType(CellType.Value);
284 
285 		for (int i = 0; i < context.getColumnCount(); i++) {
286 			Cell cell = null;
287 
288 			TableHeaderNode columnNode = columnRoot.getLeafNodeAtColIndex(i);
289 
290 			if (columnNode != null && columnNode.getPosition() != null
291 					&& columnNode.getPosition().getOrdinal() != -1
292 					&& rowNode.getPosition() != null
293 					&& rowNode.getPosition().getOrdinal() != -1) {
294 				cell = context.getCellSet().getCell(columnNode.getPosition(),
295 						rowNode.getPosition());
296 			}
297 
298 			context.setCell(cell);
299 			context.setHierarchy(null);
300 			context.setMember(null);
301 			context.setAxis(null);
302 			context.setColIndex(context.getRowHeaderCount() + i);
303 			context.setColSpan(1);
304 			context.setRowSpan(1);
305 
306 			if (columnNode == null) {
307 				Aggregator aggregator = rowNode.getAggregator();
308 				context.setAggregator(aggregator);
309 
310 				if (aggregator != null) {
311 					context.setAxis(Axis.ROWS);
312 				}
313 
314 				context.setColumnPosition(null);
315 			} else {
316 				if (columnNode.getAggregator() == null) {
317 					Aggregator aggregator = rowNode.getAggregator();
318 					context.setAggregator(aggregator);
319 
320 					if (aggregator != null) {
321 						context.setAxis(Axis.ROWS);
322 					}
323 				} else if (rowNode.getAggregator() != null
324 						&& rowNode.getAggregator().getMeasure() != null) {
325 					context.setAggregator(rowNode.getAggregator());
326 				} else {
327 					Aggregator aggregator = columnNode.getAggregator();
328 					context.setAggregator(aggregator);
329 
330 					if (aggregator != null && rowNode.getAggregator() == null) {
331 						context.setAxis(Axis.COLUMNS);
332 					}
333 				}
334 
335 				context.setColumnPosition(columnNode.getPosition());
336 			}
337 
338 			context.setRowPosition(rowNode.getPosition());
339 
340 			callback.startCell(context);
341 			callback.cellContent(context);
342 			callback.endCell(context);
343 
344 			if (rowNode.getMemberChildren() == 0) {
345 				List<Aggregator> rowAggregators = rowRoot.getReference()
346 						.getAggregators();
347 				for (Aggregator aggregator : rowAggregators) {
348 					if (context.getAggregator() != aggregator) {
349 						aggregator.aggregate(context);
350 					}
351 				}
352 			}
353 
354 			if (columnNode == null || columnNode.getMemberChildren() == 0) {
355 				List<Aggregator> columnAggregators = columnRoot.getReference()
356 						.getAggregators();
357 				for (Aggregator aggregator : columnAggregators) {
358 					if (context.getAggregator() != aggregator) {
359 						aggregator.aggregate(context);
360 					}
361 				}
362 			}
363 		}
364 
365 		context.setAggregator(null);
366 	}

368 	/**
369 	 * @param context
370 	 * @param columnRoot
371 	 * @param rowRoot
372 	 * @param callback
373 	 */
374 	protected void renderHeaderCorner(RenderContext context,
375 			TableHeaderNode columnRoot, TableHeaderNode rowRoot,
376 			PivotLayoutCallback callback) {
377 		boolean showParentMembers = context.getRenderer()
378 				.getShowParentMembers();
379 		boolean showDimensionTitle = context.getRenderer()
380 				.getShowDimensionTitle();
381 
382 		int offset = 0;
383 
384 		if (context.getRenderer().getShowDimensionTitle()) {
385 			offset = showParentMembers ? 2 : 1;
386 		}
387 
388 		context.setAxis(null);
389 
390 		context.setHierarchy(null);
391 		context.setLevel(null);
392 		context.setMember(null);
393 		context.setProperty(null);
394 
395 		context.setCell(null);
396 		context.setCellType(CellType.None);
397 
398 		context.setColumnPosition(null);
399 		context.setRowPosition(null);
400 
401 		boolean renderDimensionTitle = showDimensionTitle
402 				&& (context.getRowIndex() == context.getColumnHeaderCount()
403 						- offset);
404 		boolean renderLevelTitle = showDimensionTitle
405 				&& showParentMembers
406 				&& (context.getRowIndex() == context.getColumnHeaderCount() - 1);
407 
408 		if (context.getRowIndex() == 0 && !renderDimensionTitle
409 				&& !renderLevelTitle) {
410 			context.setColSpan(context.getRowHeaderCount());
411 			context.setRowSpan(context.getColumnHeaderCount() - offset);
412 
413 			callback.startCell(context);
414 			callback.cellContent(context);
415 			callback.endCell(context);
416 		} else if (renderDimensionTitle) {
417 			final Map<Hierarchy, Integer> spans = new HashMap<Hierarchy, Integer>();
418 			final Map<Hierarchy, List<Property>> propertyMap = new HashMap<Hierarchy, List<Property>>();
419 
420 			rowRoot.walkChildrenAtColIndex(
421 					new TreeNodeCallback<TableAxisContext>() {
422 
423 						@Override
424 						public int handleTreeNode(
425 								TreeNode<TableAxisContext> node) {
426 							TableHeaderNode headerNode = (TableHeaderNode) node;
427 							if (headerNode.getHierarchy() == null) {
428 								return TreeNodeCallback.CONTINUE;
429 							}
430 
431 							Hierarchy hierarchy = headerNode.getHierarchy();
432 
433 							if (headerNode.getProperty() == null) {
434 								Integer span = spans.get(hierarchy);
435 								if (span == null) {
436 									span = 0;
437 								}
438 
439 								span += headerNode.getRowSpan();
440 								spans.put(headerNode.getHierarchy(), span);
441 							} else {
442 								List<Property> properties = propertyMap
443 										.get(hierarchy);
444 								if (properties == null) {
445 									properties = new ArrayList<Property>();
446 									propertyMap.put(hierarchy, properties);
447 								}
448 
449 								properties.add(headerNode.getProperty());
450 							}
451 
452 							return TreeNodeCallback.CONTINUE;
453 						}
454 					}, 0);
455 
456 			context.setAxis(Axis.ROWS);
457 			context.setRowSpan(1);
458 			context.setCellType(CellType.Title);
459 
460 			for (Hierarchy hierarchy : rowRoot.getReference().getHierarchies()) {
461 				Integer span = spans.get(hierarchy);
462 				if (span == null) {
463 					span = 1;
464 				}
465 
466 				context.setColSpan(span);
467 				context.setHierarchy(hierarchy);
468 
469 				callback.startCell(context);
470 				callback.cellContent(context);
471 				callback.endCell(context);
472 
473 				context.setColIndex(context.getColIndex() + span);
474 
475 				List<Property> properties = propertyMap.get(hierarchy);
476 				if (properties != null) {
477 					for (Property property : properties) {
478 						context.setColSpan(1);
479 						context.setColIndex(context.getColIndex() + 1);
480 						context.setProperty(property);
481 
482 						callback.startCell(context);
483 						callback.cellContent(context);
484 						callback.endCell(context);
485 					}
486 				}
487 			}
488 		} else if (renderLevelTitle) {
489 			final Map<Integer, Level> levels = new HashMap<Integer, Level>();
490 			final Map<Integer, Property> properties = new HashMap<Integer, Property>();
491 
492 			rowRoot.walkChildren(new TreeNodeCallback<TableAxisContext>() {
493 
494 				@Override
495 				public int handleTreeNode(TreeNode<TableAxisContext> node) {
496 					TableHeaderNode headerNode = (TableHeaderNode) node;
497 					int colIndex = headerNode.getRowIndex() - 1;
498 
499 					if (headerNode.getMember() != null
500 							&& !levels.containsKey(colIndex)) {
501 						levels.put(colIndex, headerNode.getMember().getLevel());
502 					}
503 
504 					if (headerNode.getProperty() != null
505 							&& !properties.containsKey(colIndex)) {
506 						properties.put(colIndex, headerNode.getProperty());
507 					}
508 
509 					return TreeNodeCallback.CONTINUE;
510 				}
511 			});
512 
513 			context.setAxis(Axis.ROWS);
514 			context.setColSpan(1);
515 			context.setRowSpan(1);
516 			context.setCellType(CellType.Title);
517 
518 			for (int i = 0; i < context.getRowHeaderCount(); i++) {
519 				context.setColIndex(i);
520 
521 				Level level = levels.get(i);
522 
523 				if (level == null) {
524 					context.setHierarchy(null);
525 					context.setLevel(null);
526 				} else {
527 					context.setHierarchy(level.getHierarchy());
528 					context.setLevel(level);
529 				}
530 
531 				context.setProperty(properties.get(i));
532 
533 				callback.startCell(context);
534 				callback.cellContent(context);
535 				callback.endCell(context);
536 			}
537 		}
538 
539 		context.setHierarchy(null);
540 	}

542 	/**
543 	 * @param model
544 	 * @param renderer
545 	 * @param axis
546 	 * @return
547 	 */
548 	protected TableHeaderNode createAxisTree(PivotModel model,
549 			PivotRenderer renderer, Axis axis) {
550 		CellSetAxis cellSetAxis = model.getCellSet().getAxes()
551 				.get(axis.axisOrdinal());
552 
553 		List<Position> positions = cellSetAxis.getPositions();
554 		if (positions == null || positions.isEmpty()) {
555 			return null;
556 		}
557 
558 		List<Hierarchy> hierarchies = new ArrayList<Hierarchy>();
559 		List<Aggregator> aggregators = new ArrayList<Aggregator>();
560 
561 		Map<Hierarchy, List<Level>> levelsMap = new HashMap<Hierarchy, List<Level>>();
562 
563 		Comparator<Level> levelComparator = new Comparator<Level>() {
564 
565 			@Override
566 			public int compare(Level l1, Level l2) {
567 				Integer d1 = l1.getDepth();
568 				Integer d2 = l2.getDepth();
569 				return d1.compareTo(d2);
570 			}
571 		};
572 
573 		boolean containsMeasure = false;
574 
575 		List<Member> firstMembers = positions.get(0).getMembers();
576 
577 		int index = 0;
578 		for (Member member : firstMembers) {
579 			if (member instanceof Measure) {
580 				containsMeasure = true;
581 				break;
582 			}
583 
584 			index++;
585 		}
586 
587 		AggregatorFactory aggregatorFactory = renderer.getAggregatorFactory();
588 
589 		List<String> aggregatorNames = null;
590 		List<String> hierarchyAggregatorNames = null;
591 		List<String> memberAggregatorNames = null;
592 
593 		if (aggregatorFactory != null
594 				&& (!containsMeasure || index == firstMembers.size() - 1)) {
595 			aggregatorNames = renderer.getAggregators(axis,
596 					AggregatorPosition.Grand);
597 			hierarchyAggregatorNames = renderer.getAggregators(axis,
598 					AggregatorPosition.Hierarchy);
599 			memberAggregatorNames = renderer.getAggregators(axis,
600 					AggregatorPosition.Member);
601 		}
602 
603 		if (aggregatorNames == null) {
604 			aggregatorNames = Collections.emptyList();
605 		}
606 
607 		if (hierarchyAggregatorNames == null) {
608 			hierarchyAggregatorNames = Collections.emptyList();
609 		}
610 
611 		if (memberAggregatorNames == null) {
612 			memberAggregatorNames = Collections.emptyList();
613 		}
614 
615 		TableAxisContext nodeContext = new TableAxisContext(axis, hierarchies,
616 				levelsMap, aggregators, renderer);
617 
618 		TableHeaderNode axisRoot = new TableHeaderNode(nodeContext);
619 
620 		Set<Measure> grandTotalMeasures = new LinkedHashSet<Measure>();
621 		Set<Measure> totalMeasures = new LinkedHashSet<Measure>();
622 
623 		List<Member> lastMembers = null;
624 
625 		Map<Hierarchy, List<AggregationTarget>> memberParentMap = new HashMap<Hierarchy, List<AggregationTarget>>();
626 
627 		for (Position position : positions) {
628 			TableHeaderNode lastChild = null;
629 
630 			List<Member> members = position.getMembers();
631 
632 			int memberCount = members.size();
633 
634 			for (int i = memberCount - 1; i >= 0; i--) {
635 				Member member = members.get(i);
636 
637 				if (member instanceof Measure) {
638 					Measure measure = (Measure) member;
639 
640 					if (!totalMeasures.contains(measure)) {
641 						totalMeasures.add(measure);
642 					}
643 
644 					if (!grandTotalMeasures.contains(measure)) {
645 						grandTotalMeasures.add(measure);
646 					}
647 				}
648 
649 				if (hierarchies.size() < memberCount) {
650 					hierarchies.add(0, member.getHierarchy());
651 				}
652 
653 				List<Level> levels = levelsMap.get(member.getHierarchy());
654 
655 				if (levels == null) {
656 					levels = new ArrayList<Level>();
657 					levelsMap.put(member.getHierarchy(), levels);
658 				}
659 
660 				if (!levels.contains(member.getLevel())) {
661 					levels.add(0, member.getLevel());
662 					Collections.sort(levels, levelComparator);
663 				}
664 
665 				TableHeaderNode childNode = new TableHeaderNode(nodeContext);
666 				childNode.setMember(member);
667 				childNode.setHierarchy(member.getHierarchy());
668 				childNode.setPosition(position);
669 
670 				if (lastChild != null) {
671 					childNode.addChild(lastChild);
672 				}
673
