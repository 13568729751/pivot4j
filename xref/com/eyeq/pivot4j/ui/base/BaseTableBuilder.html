45  		String label = null;
46  
47  		if (context.getCell() != null) {
48  			label = context.getCell().getFormattedValue();
49  		} else if (context.getMember() != null) {
50  			label = context.getMember().getCaption();
51  		} else if (context.getHierarchy() != null) {
52  			label = context.getHierarchy().getDimension().getCaption();
53  		}
54

56  		cell.setLabel(label);
