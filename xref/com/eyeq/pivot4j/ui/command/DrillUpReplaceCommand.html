23  public class DrillUpReplaceCommand extends AbstractDrillDownCommand {
24  
25  	public static final String NAME = "drillUp";
26  
27  	/**
28  	 * @param renderer
29  	 */

31  		super(renderer);
32  	}
33  
34  	/**
35  	 * @see com.eyeq.pivot4j.ui.command.CellCommand#getName()
36  	 */
37  	@Override
38  	public String getName() {
39  		return NAME;
40  	}
41  
42  	/**
43  	 * @see com.eyeq.pivot4j.ui.command.AbstractCellCommand#getMode()
44  	 */
45  	@Override
46  	public String getMode() {
47  		return MODE_REPLACE;
48  	}
49  
50  	/**
51  	 * @see com.eyeq.pivot4j.ui.command.CellCommand#canExecute(com.eyeq.pivot4j.ui
52  	 *      .RenderContext)
53  	 */
54  	@Override
55  	public boolean canExecute(RenderContext context) {
56  		if (!super.canExecute(context)) {
57  			return false;
58  		}
59  
60  		PivotModel model = context.getModel();
61  
62  		DrillReplace transform = model.getTransform(DrillReplace.class);
63  
64  		Hierarchy hierarchy = context.getHierarchy();
65  		if (hierarchy == null) {
66  			return false;
67  		}
68  
69  		return transform.canDrillUp(hierarchy);
70  	}
71  
72  	/**
73  	 * @see com.eyeq.pivot4j.ui.command.CellCommand#createParameters(com.eyeq.pivot4j
74  	 *      .ui.RenderContext)
75  	 */
76  	@Override
77  	public CellParameters createParameters(RenderContext context) {
78  		CellParameters parameters = new CellParameters();
79  		parameters.setAxisOrdinal(context.getAxis().axisOrdinal());
80  		parameters.setHierarchyUniqueName(context.getHierarchy()
81  				.getUniqueName());
82  
83  		return parameters;
84  	}
85  
86  	/**
87  	 * @see com.eyeq.pivot4j.ui.command.CellCommand#execute(com.eyeq.pivot4j.PivotModel
88  	 *      , com.eyeq.pivot4j.ui.command.CellParameters)
89  	 */
90  	@Override
91  	public void execute(PivotModel model, CellParameters parameters) {
92  		CellSet cellSet = model.getCellSet();
93  
94  		CellSetAxis axis = cellSet.getAxes().get(parameters.getAxisOrdinal());
95  
96  		List<Member> members = axis.getPositions().get(0).getMembers();
97  
98  		Hierarchy hierarchy = null;
99  		for (Member m : members) {
100 			if (m.getHierarchy().getUniqueName()
101 					.equals(parameters.getHierarchyUniqueName())) {
102 				hierarchy = m.getHierarchy();
103 				break;
104 			}
105 		}
106 
107 		DrillReplace transform = model.getTransform(DrillReplace.class);
108 		transform.drillUp(hierarchy);
109 	}
