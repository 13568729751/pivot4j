9   package com.eyeq.pivot4j.ui.command;
10  
11  import org.olap4j.CellSet;
12  import org.olap4j.CellSetAxis;
13  import org.olap4j.Position;
14  import org.olap4j.metadata.Member;
15  
16  import com.eyeq.pivot4j.PivotModel;
17  import com.eyeq.pivot4j.transform.DrillExpandPosition;
18  import com.eyeq.pivot4j.ui.RenderContext;
19  
20  public class DrillCollapsePositionCommand implements DrillDownCommand {
21  
22  	public static final String NAME = "collapsePosition";
23  
24  	/**
25  	 * @see com.eyeq.pivot4j.ui.command.CellCommand#getName()
26  	 */
27  	@Override
28  	public String getName() {
29  		return NAME;
30  	}
31  
32  	/**
33  	 * @see com.eyeq.pivot4j.ui.command.CellCommand#canExecute(com.eyeq.pivot4j.ui
34  	 *      .RenderContext)
35  	 */
36  	@Override
37  	public boolean canExecute(RenderContext context) {
38  		PivotModel model = context.getModel();
39  
40  		DrillExpandPosition transform = model
41  				.getTransform(DrillExpandPosition.class);
42  
43  		Position position = context.getPosition();
44  		Member member = context.getMember();
45  
46  		if (position == null || member == null) {
47  			return false;
48  		}
49  
50  		return transform.canCollapse(position, member);
51  	}
52  
53  	/**
54  	 * @see com.eyeq.pivot4j.ui.command.CellCommand#createParameters(com.eyeq.pivot4j
55  	 *      .ui.RenderContext)
56  	 */
57  	@Override
58  	public CellParameters createParameters(RenderContext context) {
59  		CellParameters parameters = new CellParameters();
60  		parameters.setAxisOrdinal(context.getAxis().axisOrdinal());
61  		parameters.setMemberUniqueName(context.getMember().getUniqueName());
62  		parameters.setPositionOrdinal(context.getPosition().getOrdinal());
63  
64  		return parameters;
65  	}
66  
67  	/**
68  	 * @see com.eyeq.pivot4j.ui.command.CellCommand#execute(com.eyeq.pivot4j.PivotModel
69  	 *      , com.eyeq.pivot4j.ui.command.CellParameters)
70  	 */
71  	@Override
72  	public void execute(PivotModel model, CellParameters parameters) {
73  		CellSet cellSet = model.getCellSet();
74  
75  		CellSetAxis axis = cellSet.getAxes().get(parameters.getAxisOrdinal());
76  		Position position = axis.getPositions().get(
77  				parameters.getPositionOrdinal());
78  
79  		Member member = null;
80  
81  		for (Member m : position.getMembers()) {
82  			if (m.getUniqueName().equals(parameters.getMemberUniqueName())) {
83  				member = m;
84  				break;
85  			}
86  		}
87  
88  		DrillExpandPosition transform = model
89  				.getTransform(DrillExpandPosition.class);
90  		transform.collapse(position, member);
91  	}
92  }
