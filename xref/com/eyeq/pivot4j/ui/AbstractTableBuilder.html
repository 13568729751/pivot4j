83  	/**
84  	 * @see com.eyeq.pivot4j.ui.TableBuilder#build(com.eyeq.pivot4j.PivotModel)
85  	 */
86  	@Override
87  	public T build(PivotModel model) {
88  		if (model == null) {
89  			throw new IllegalArgumentException(
90  					"Missing required argument 'model'.");
91  		}
92  
93  		BuildContext context = createBuildContext(model);
94  
95  		T table = createTable(context);
96  
97  		HeadersInfo columnHeader = createColumnHeaders(context, table);
98  		if (columnHeader != null) {
99  			table.getHeaders().addAll(columnHeader.getHeaders());
100 		}
101 
102 		HeadersInfo rowHeader = createRowHeaders(context, table);
103 		if (rowHeader != null) {
104 			table.getRows().addAll(rowHeader.getHeaders());
105 		}
106 
107 		if (columnHeader != null && rowHeader != null) {
108 			populateCells(context, table, columnHeader, rowHeader);
109 
110 			addCornerCells(context, table, columnHeader, rowHeader);
111 		}
112 
113 		return table;
114 	}

122 		List<CellSetAxis> axes = context.getCellSet().getAxes();
123 		if (axes.isEmpty()) {
124 			return null;
125 		}
126 
127 		CellSetAxis axis = axes.get(Axis.COLUMNS.axisOrdinal());
128 		context.setAxis(axis);
129 
130 		HeaderNode axisRoot = createAxisTree(axis);
131 
132 		HeadersInfo headers = createColumnHeaders(context, table, axisRoot);
133 
134 		context.setAxis(null);
135 
136 		return headers;

139 	/**
140 	 * @param context
141 	 * @param table
142 	 * @param axisRoot
143 	 * @return
144 	 */
145 	protected HeadersInfo createColumnHeaders(BuildContext context, T table,
146 			HeaderNode axisRoot) {
147 		CellSetAxis axis = context.getAxis();
148 
149 		List<Position> positions = axis.getPositions();
150 		if (positions == null || positions.isEmpty()) {
151 			return null;
152 		}
153 
154 		List<TR> headers = new ArrayList<TR>();
155 
156 		List<Member> members = positions.get(0).getMembers();
157 		List<Hierarchy> hierarchies = new ArrayList<Hierarchy>(members.size());
158 		List<Integer> spans = new ArrayList<Integer>(members.size());
159 
160 		int headerIndex = 0;
161 		for (Member member : members) {
162 			Hierarchy hierarchy = member.getHierarchy();
163 
164 			hierarchies.add(hierarchy);
165 
166 			context.setHierarchy(hierarchy);
167 
168 			List<HeaderNode> nodes = axisRoot.getNodesForHierarchy(hierarchy);
169 
170 			List<TR> hierarchyHeaders = createColumnHeadersForHierarchy(
171 					context, table, headerIndex, nodes);
172 			if (hierarchyHeaders != null) {
173 				headers.addAll(hierarchyHeaders);
174 
175 				int span = hierarchyHeaders.size();
176 				headerIndex += span;
177 				spans.add(span);
178 			}
179 		}
180 
181 		context.setHierarchy(null);
182 
183 		return new HeadersInfo(headers, hierarchies, spans);
184 	}

186 	/**
187 	 * @param context
188 	 * @param table
189 	 * @param headerIndex
190 	 * @param children
191 	 * @return
192 	 */
193 	protected List<TR> createColumnHeadersForHierarchy(BuildContext context,
194 			T table, int headerIndex, List<HeaderNode> children) {
195 		List<TR> headers = new ArrayList<TR>();
196 
197 		if (getShowDimensionTitle()) {
198 			context.setColumnPosition(null);
199 			context.setMember(null);
200 
201 			TR row = createRow(context, table, headerIndex);
202 			headers.add(row);
203 
204 			Member parent = null;
205 
206 			int colIndex = 0;
207 			int colSpan = 0;
208 
209 			for (HeaderNode node : children) {
210 				if (parent != null
211 						&& !ObjectUtils.equals(parent, node.getParent()
212 								.getReference())) {
213 					addSpanCells(context, table, row, headerIndex, colIndex,
214 							colSpan, 1);
215 
216 					parent = node.getParent().getReference();
217 
218 					colIndex += colSpan;
219 					colSpan = 0;
220 				}
221 
222 				colSpan += node.getWidth();
223 				parent = node.getParent().getReference();
224 			}
225 
226 			addSpanCells(context, table, row, headerIndex, colIndex, colSpan, 1);
227 
228 			headerIndex++;
229 		}
230 
231 		int minDepth = 0;
232 		int maxSpan = 1;
233 
234 		if (showParentMembers) {
235 			for (HeaderNode node : children) {
236 				if (node.isSiblingParent()) {
237 					int depth = node.getMaxSiblingDepth()
238 							- node.getReference().getDepth() + 1;
239 					maxSpan = Math.max(maxSpan, depth);
240 				}
241 
242 				minDepth = Math.min(minDepth, node.getReference().getDepth());
243 			}
244 		}
245 
246 		int colIndex = 0;
247 
248 		for (HeaderNode node : children) {
249 			context.setMember(node.getReference());
250 			context.setColumnPosition(node.getPosition());
251 
252 			TR row = null;
253 
254 			int nodeWidth = node.getWidth();
255 
256 			int colSpan = nodeWidth;
257 			int rowSpan;
258 
259 			int offset;
260 
261 			if (showParentMembers) {
262 				offset = node.getReference().getDepth() - minDepth;
263 				rowSpan = maxSpan - offset;
264 			} else {
265 				rowSpan = 1;
266 				offset = 0;
267 			}
268 
269 			int rowIndex = headerIndex + offset;
270 
271 			if (showDimensionTitle) {
272 				offset++;
273 			}
274 
275 			if (headers.size() > offset) {
276 				row = headers.get(offset);
277 			} else {
278 				row = createRow(context, table, rowIndex);
279 				headers.add(row);
280 			}
281 
282 			if (showParentMembers && node.isSiblingParent()) {
283 				addSpanCells(context, table, row, rowIndex, colIndex, colSpan,
284 						rowSpan);
285 
286 				int siblingWidth = 0;
287 				for (HeaderNode siblingChild : node.getSiblingChildren()) {
288 					siblingWidth += siblingChild.getWidth();
289 					if (siblingChild.isSiblingParent()) {
290 						siblingWidth += siblingChild.getSiblingWidth();
291 					}
292 				}
293 
294 				addSpanCells(context, table, row, rowIndex, colIndex,
295 						siblingWidth, 1);
296 			} else {
297 				addSpanCells(context, table, row, rowIndex, colIndex, colSpan,
298 						rowSpan);
299 			}
300 
301 			colIndex += colSpan;
302 		}
303 
304 		context.setMember(null);
305 		context.setColumnPosition(null);
306 
307 		return headers;
308 	}

333 	/**
334 	 * @param context
335 	 * @param table
336 	 * @param axisRoot
337 	 * @return
338 	 */
339 	protected HeadersInfo createRowHeaders(final BuildContext context,
340 			final T table, final HeaderNode axisRoot) {
341 		final List<TR> rows = new ArrayList<TR>();
342 
343 		CellSetAxis axis = context.getAxis();
344 
345 		List<Position> positions = axis.getPositions();
346 		if (positions == null || positions.isEmpty()) {
347 			return null;
348 		}
349 
350 		List<Member> members = positions.get(0).getMembers();
351 		final List<Hierarchy> hierarchies = new ArrayList<Hierarchy>(
352 				members.size());
353 		for (Member member : members) {
354 			hierarchies.add(member.getHierarchy());
355 		}
356 
357 		final List<Integer> spans = new ArrayList<Integer>(members.size());
358 
359 		final Map<Hierarchy, Integer> maxSpanMap = new HashMap<Hierarchy, Integer>(
360 				members.size());
361 		final Map<Hierarchy, Integer> minDepthMap = new HashMap<Hierarchy, Integer>(
362 				members.size());
363 
364 		axisRoot.walkChildren(new TreeNodeCallback<Member>() {
365 
366 			@Override
367 			public int handleTreeNode(TreeNode<Member> node) {
368 				HeaderNode memberNode = (HeaderNode) node;
369 				Hierarchy hierarchy = memberNode.getReference().getHierarchy();
370 
371 				Integer maxSpan = maxSpanMap.get(hierarchy);
372 				if (maxSpan == null) {
373 					maxSpan = 1;
374 				}
375 
376 				if (memberNode.isSiblingParent()) {
377 					int span = memberNode.getMaxSiblingDepth()
378 							- node.getReference().getDepth() + 1;
379 					maxSpan = Math.max(maxSpan, span);
380 				}
381 
382 				maxSpanMap.put(hierarchy, maxSpan);
383 
384 				Integer minDepth = minDepthMap.get(hierarchy);
385 				if (minDepth == null) {
386 					minDepth = 0;
387 				}
388 
389 				minDepthMap.put(hierarchy,
390 						Math.min(minDepth, node.getReference().getDepth()));
391 
392 				return CONTINUE;
393 			}
394 		});
395 
396 		axisRoot.walkChildren(new TreeNodeCallback<Member>() {
397 
398 			int colIndex = 0;
399 			int rowIndex = 0;
400 
401 			@Override
402 			public int handleTreeNode(TreeNode<Member> node) {
403 				HeaderNode memberNode = (HeaderNode) node;
404 
405 				Member member = node.getReference();
406 				Hierarchy hierarchy = member.getHierarchy();
407 				Position position = memberNode.getPosition();
408 
409 				context.setMember(member);
410 				context.setRowPosition(position);
411 
412 				int maxSpan = maxSpanMap.get(hierarchy);
413 				int minDepth = minDepthMap.get(hierarchy);
414 
415 				int hierarchyIndex = node.getLevel() - 1;
416 
417 				if (spans.size() <= hierarchyIndex) {
418 					spans.add(maxSpan);
419 				}
420 
421 				int rowSpan = memberNode.getWidth();
422 				int colSpan;
423 
424 				if (showParentMembers) {
425 					int baseColIndex = 0;
426 
427 					for (Hierarchy hier : hierarchies) {
428 						if (hier.equals(hierarchy)) {
429 							break;
430 						}
431 						baseColIndex += maxSpanMap.get(hier);
432 					}
433 
434 					colIndex = baseColIndex + member.getDepth() - minDepth;
435 					colSpan = maxSpan - member.getDepth() + minDepth;
436 				} else {
437 					colSpan = 1;
438 					colIndex = hierarchyIndex;
439 				}
440 
441 				addSpanCells(context, table, rows, rowIndex, colIndex, colSpan,
442 						rowSpan);
443 
444 				if (showParentMembers && memberNode.isSiblingParent()) {
445 					int siblingWidth = 0;
446 					for (HeaderNode siblingChild : memberNode
447 							.getSiblingChildren()) {
448 						siblingWidth += siblingChild.getWidth();
449 						if (siblingChild.isSiblingParent()) {
450 							siblingWidth += siblingChild.getSiblingWidth();
451 						}
452 					}
453 
454 					while (rows.size() <= rowIndex + rowSpan) {
455 						TR row = createRow(context, table, rows.size());
456 						rows.add(row);
457 					}
458 
459 					addSpanCells(context, table, rows, rowIndex + rowSpan,
460 							colIndex, 1, siblingWidth);
461 				}
462 
463 				if (node.getChildren() == null || node.getChildren().isEmpty()) {
464 					colIndex = 0;
465 					rowIndex++;
466 				}
467 
468 				context.setMember(null);
469 				context.setRowPosition(null);
470 
471 				return CONTINUE;
472 			}
473 		});
474 
475 		context.setHierarchy(null);
476 
477 		return new HeadersInfo(rows, hierarchies, spans);
478 	}

480 	/**
481 	 * @param context
482 	 * @param table
483 	 * @param columnHeader
484 	 * @param rowHeader
485 	 */
486 	protected void addCornerCells(BuildContext context, T table,
487 			HeadersInfo columnHeader, HeadersInfo rowHeader) {
488 		TR firstHeader = columnHeader.getHeaders().get(0);
489 
490 		int width = rowHeader.getWidth();
491 		int height = columnHeader.getHeight();
492 
493 		if (showDimensionTitle) {
494 			height--;
495 		}
496 
497 		TC corner = createCell(context, table, firstHeader, 0, 0, width, height);
498 		firstHeader.getCells().add(0, corner);
499 
500 		if (showDimensionTitle) {
501 			context.setAxis(context.getCellSet().getAxes()
502 					.get(Axis.ROWS.axisOrdinal()));
503 
504 			int colIndex = 0;
505 			int hierarchyIndex = 0;
506 
507 			List<Hierarchy> hierarchies = rowHeader.getHierarchies();
508 
509 			List<TC> cells = new ArrayList<TC>(hierarchies.size());
510 			for (Hierarchy hierarchy : hierarchies) {
511 				context.setHierarchy(hierarchy);
512 
513 				int span;
514 
515 				if (showParentMembers) {
516 					span = rowHeader.getSpans().get(hierarchyIndex);
517 				} else {
518 					span = 1;
519 				}
520 
521 				TC header = createCell(context, table, firstHeader, colIndex,
522 						height - 1, span, 1);
523 				cells.add(header);
524 
525 				hierarchyIndex++;
526 				colIndex += span;
527 			}
528 
529 			Collections.reverse(cells);
530 
531 			TR lastHeader = columnHeader.getHeaders().get(
532 					columnHeader.getHeaders().size() - 1);
533 
534 			for (TC cell : cells) {
535 				lastHeader.getCells().add(0, cell);
536 			}
537 
538 			context.setHierarchy(null);
539 			context.setAxis(null);
540 		}
541 	}

543 	/**
544 	 * @param context
545 	 * @param table
546 	 * @param columnHeader
547 	 * @param rowHeader
548 	 */
549 	protected void populateCells(BuildContext context, T table,
550 			HeadersInfo columnHeader, HeadersInfo rowHeader) {
551 		int width = columnHeader.getWidth();
552 		int height = rowHeader.getHeight();
553 
554 		List<TR> headers = rowHeader.getHeaders();
555 
556 		int ordinal = 0;
557 		for (int rowIndex = 0; rowIndex < height; rowIndex++) {
558 			TR row = headers.get(rowIndex);
559 
560 			for (int colIndex = 0; colIndex < width; colIndex++) {
561 				Cell cell = context.getCellSet().getCell(ordinal++);
562 				context.setCell(cell);
563 
564 				TC tableCell = createCell(context, table, row, colIndex,
565 						rowIndex, 1, 1);
566 				row.getCells().add(tableCell);
567 			}
568 		}
569 
570 		context.setCell(null);
571 	}

595 	/**
596 	 * @param context
597 	 * @param table
598 	 * @param row
599 	 * @param colIndex
600 	 * @param rowIndex
601 	 * @param colSpan
602 	 * @param rowSpan
603 	 * @return
604 	 */
605 	protected TC createCell(BuildContext context, T table, TR row,
606 			int colIndex, int rowIndex, int colSpan, int rowSpan) {
607 		CellType type = null;
608 
609 		CellSetAxis axis = context.getAxis();
610 
611 		if (context.getCell() != null) {
612 			type = CellType.Value;
613 		} else if (context.getMember() != null) {
614 			if (axis.getAxisOrdinal() == Axis.ROWS) {
615 				type = CellType.RowHeader;
616 			} else if (axis.getAxisOrdinal() == Axis.COLUMNS) {
617 				type = CellType.ColumnHeader;
618 			}
619 		} else if (context.getHierarchy() != null) {
620 			if (axis.getAxisOrdinal() == Axis.ROWS) {
621 				type = CellType.RowTitle;
622 			} else if (axis.getAxisOrdinal() == Axis.COLUMNS) {
623 				type = CellType.ColumnTitle;
624 			}
625 		} else {
626 			type = CellType.None;
627 		}
628 
629 		TC cell = createCell(context, table, row, type, colIndex, rowIndex,
630 				colSpan, rowSpan);
631 		configureCell(context, table, row, colIndex, rowIndex, cell);
632 
633 		return cell;
634 	}

654 	/**
655 	 * @param context
656 	 * @param table
657 	 * @param row
658 	 * @param colIndex
659 	 * @param rowIndex
660 	 * @param cell
661 	 * @return
662 	 */
663 	protected String getCellLabel(BuildContext context, T table, TR row,
664 			int colIndex, int rowIndex, TC cell) {
665 		String label = null;
666 
667 		switch (cell.getType()) {
668 		case Value:
669 			label = context.getCell().getFormattedValue();
670 			break;
671 		case ColumnHeader:
672 		case RowHeader:
673 			label = context.getMember().getCaption();
674 			break;
675 		case ColumnTitle:
676 		case RowTitle:
677 			label = context.getHierarchy().getDimension().getCaption();
678 			break;
679 		case None:
680 			break;
681 		default:
682 			assert false;
683 		}
684 
685 		return label;
686 	}

697 	private void addSpanCells(BuildContext context, T table, TR row,
698 			int rowIndex, int colIndex, int colSpan, int rowSpan) {
699 		if (hideSpans) {
700 			for (int i = 0; i < colSpan; i++) {
701 				TC cell = createCell(context, table, row, colIndex + i,
702 						rowIndex, 1, rowSpan);
703 				row.getCells().add(cell);
704 			}
705 		} else {
706 			TC cell = createCell(context, table, row, colIndex, rowIndex,
707 					colSpan, rowSpan);
708 			row.getCells().add(cell);
709 		}
710 	}

712 	/**
713 	 * @param context
714 	 * @param table
715 	 * @param rows
716 	 * @param rowIndex
717 	 * @param colIndex
718 	 * @param colSpan
719 	 * @param rowSpan
720 	 */
721 	private void addSpanCells(BuildContext context, T table, List<TR> rows,
722 			int rowIndex, int colIndex, int colSpan, int rowSpan) {
723 		if (hideSpans) {
724 			for (int i = 0; i < rowSpan; i++) {
725 				int index = rowIndex + i;
726 
727 				TR row;
728 
729 				if (rows.size() > index) {
730 					row = rows.get(index);
731 				} else {
732 					row = createRow(context, table, rows.size());
733 					rows.add(row);
734 				}
735 
736 				TC cell = createCell(context, table, row, colIndex, index,
737 						colSpan, 1);
738 				row.getCells().add(cell);
739 			}
740 		} else {
741 			TR row;
742 
743 			if (rows.size() > rowIndex) {
744 				row = rows.get(rowIndex);
745 			} else {
746 				row = createRow(context, table, rows.size());
747 				rows.add(row);
748 			}
749 			TC cell = createCell(context, table, row, colIndex, rowIndex,
750 					colSpan, rowSpan);
