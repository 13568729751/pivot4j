450 	@Override
451 	public void cellContent(RenderContext context) {
452 		cell.setCellStyle(getCellStyle(context));
453 
454 		if (context.getCellType() == CellType.Value
455 				&& context.getAxis() != Axis.FILTER) {
456 			Double value = null;
457 
458 			if (context.getAggregator() != null) {
459 				value = context.getAggregator().getValue(context);
460 			} else if (context.getCell() != null
461 					&& !context.getCell().isEmpty()) {
462 				try {
463 					value = context.getCell().getDoubleValue();
464 				} catch (OlapException e) {
465 					throw new PivotException(e);
466 				}
467 			}
468 
469 			cellContent(context, value);
470 		} else {
471 			super.cellContent(context);
472 		}
473 	}

538 	/**
539 	 * @param context
540 	 * @return
541 	 */
542 	protected CellRangeAddress createMergedRegion(RenderContext context) {
543 		if (context.getColumnSpan() > 1 || context.getRowSpan() > 1) {
544 			int firstRow = context.getRowIndex() + getRowOffset();
545 			int lastRow = firstRow + context.getRowSpan() - 1;
546 			int firstCol = context.getColumnIndex() + getColOffset();
547 			int lastCol = firstCol + context.getColumnSpan() - 1;
548 
549 			return new CellRangeAddress(firstRow, lastRow, firstCol, lastCol);
550 		} else {
551 			return null;
552 		}
553 	}

568 	 * @param context
569 	 * @param sheet
570 	 */
571 	protected void adjustColumnSizes(RenderContext context, Sheet sheet) {
572 		for (int i = 0; i < context.getColumnCount(); i++) {
573 			try {
574 				sheet.autoSizeColumn(getColOffset() + i,
575 						!mergedRegions.isEmpty());
576 			} catch (Exception e) {
577 				// Ignore any problem while calculating size of the columns.
578 			}
579 		}
580 	}
