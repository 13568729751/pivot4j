9   package com.eyeq.pivot4j.query;

11  import org.olap4j.metadata.Hierarchy;
12  
13  import com.eyeq.pivot4j.mdx.AbstractExp;
14  import com.eyeq.pivot4j.mdx.Exp;
15  import com.eyeq.pivot4j.mdx.ExpVisitor;

17  /**
18   * Wrapper for Set Object to be placed on a query axis
19   */
20  public class SetExp extends AbstractExp {
21  
22  	private static final long serialVersionUID = 1634345220637156479L;
23  
24  	private CalcSetMode mode;
25  
26  	private Exp expression;
27  
28  	private Hierarchy hierarchy;
29  
30  	/**
31  	 * @param mode
32  	 * @param expression
33  	 * @param hierarchy
34  	 */
35  	public SetExp(CalcSetMode mode, Exp expression, Hierarchy hierarchy) {
36  		this.mode = mode;
37  		this.expression = expression;
38  		this.hierarchy = hierarchy;
39  	}
40  
41  	/**
42  	 * @return
43  	 */
44  	public Hierarchy getHierarchy() {
45  		return hierarchy;
46  	}
47  
48  	/**
49  	 * @return
50  	 */
51  	public CalcSetMode getMode() {
52  		return mode;
53  	}
54  
55  	/**
56  	 * @return
57  	 */
58  	public Exp getExpression() {
59  		return expression;
60  	}
61  
62  	/**
63  	 * @see com.eyeq.pivot4j.mdx.Exp#toMdx()
64  	 */
65  	public String toMdx() {
66  		if (expression == null) {
67  			return null;
68  		}
69  
70  		return expression.toMdx();
71  	}
72  
73  	/**
74  	 * @see com.eyeq.pivot4j.mdx.Exp#accept(com.eyeq.pivot4j.mdx.ExpVisitor)
75  	 */
76  	public void accept(ExpVisitor visitor) {
77  		if (expression != null) {
78  			expression.accept(visitor);
79  		}
80  	}
81  
82  	/**
83  	 * @see com.eyeq.pivot4j.mdx.Exp#copy()
84  	 */
85  	public SetExp copy() {
86  		return new SetExp(mode, expression.copy(), hierarchy);
87  	}
88  }
